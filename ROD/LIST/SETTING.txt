# 全ての変数の行には空白を含まない、コメントを追加する際には行頭に#を入れる。

# MAX_page
# 最大検索数。この数のURLを調べたら終了する。
# 数値のみ使用可能。
MAX_page=100000

# MAX_time
# 最大実行時間。この時間内に終わらなければセーブして終了する。
# 単位は秒。数値か掛け算のみ使用可能。0は無制限。
MAX_time=3600*0

# SAVE_time
# セーブのための区切り時間。この時間経つと途中経過を保存し、再スタートする。
# MAX_time以内ならば何度でもセーブする。
# MAX_time=24時間、SAVE_time=5時間なら、4回途中保存+24時間以内に終わらなければ最後にもセーブ。
# 単位は秒。数値か掛け算のみ使用可能。0は無制限。
SAVE_time=3600*0

# run_count
# 今何回目の実行か。最初から始める場合は0。
# それ以外は途中保存したフォルダであるresult/result_*の数字を入れる。
# 数値のみ使用可能。
run_count=0

# MAX_process
# 各サーバを担当する子プロセスの最大作成数。多すぎると重くなり、結果が安定しない(接続エラー多発など)。
# 数値のみ使用可能。0は論理プロセッサ数を使用。論理プロセッサ数で余裕を持つくらい。
MAX_process=0

# screenshots
# 各ページのスクリーンショットを撮るかどうか
# TrueかFalseのみ使用可能。
screenshots=False

machine_learning=True
clamd_checking=True

# assignOrAchievement
# MAX_pageで指定した数を達成したか、割り当てたかのどちらの判断基準を使うかどうか。
# Trueは割り当て、Falseは達成。この2値以外使用不可。
# True(割り当て)にした場合、1つの子プロセスに多数が割り当てられ、終了に時間がかかる可能性が出る。
assignOrAchievement=False
